/* CALENDAR */


let selectedDates = [];


function formatDate(date) {

  return date.toISOString().split("T")[0];

}


function toggleDate(dateStr, el) {

  if (selectedDates.includes(dateStr)) {

    selectedDates = selectedDates.filter(d => d !== dateStr);

    el.classList.remove("selected");

  } else {

    selectedDates.push(dateStr);

    el.classList.add("selected");

  }

}


function renderMonth(name, start, end) {


  const container = document.getElementById("calendar");


  const title = document.createElement("h3");

  title.innerText = name + " 2026";


  const grid = document.createElement("div");

  grid.className = "calendar-grid";


  const days = ["Sun","Mon","Tue","Wed","Thu","Fri","Sat"];

  days.forEach(d => {

    const cell = document.createElement("div");

    cell.className = "day-header";

    cell.innerText = d;

    grid.appendChild(cell);

  });


  const startDate = new Date(start);

  const endDate = new Date(end);


  // add blank cells for correct alignment

  for (let i = 0; i < startDate.getDay(); i++) {

    grid.appendChild(document.createElement("div"));

  }


  let current = new Date(startDate);


  while (current <= endDate) {


    const el = document.createElement("div");

    const dateStr = formatDate(current);

    const day = current.getDay();


    el.innerText = current.getDate();


    // weekends disabled

    if (day === 0 || day === 6) {

      el.className = "day disabled";

    } else {

      el.className = "day";

      el.onclick = () => toggleDate(dateStr, el);

    }


    grid.appendChild(el);

    current.setDate(current.getDate() + 1);

  }


  container.appendChild(title);

  container.appendChild(grid);

}


renderMonth("August", "2026-08-01", "2026-08-31");

renderMonth("September", "2026-09-01", "2026-09-30");

renderMonth("October", "2026-10-01", "2026-10-31");

/* SUBMIT */


function submitData() {


  const firstName = document.getElementById("firstName").value;

  const lastName = document.getElementById("lastName").value;


  let contacts = [];


  document.querySelectorAll("#tableBody tr").forEach(row => {

    const name = row.querySelector(".name").value;

    const role = row.querySelector(".role").value;

    const email = row.querySelector(".email").value;

    const comments = row.querySelector(".comments").value;


    if (name || email) {

      contacts.push({ name, role, email, comments });

    }

  });


  const dataString = JSON.stringify({

    first_name: firstName,

    last_name: lastName,

    unavailable_dates: selectedDates.join(","),

    contacts: contacts

  });


  const form = document.createElement("form");

  form.method = "POST";

  form.action = "https://script.google.com/macros/s/AKfycby6kRP0selr_mxvaonNQLJR-sWtl845EX4qnP4rENbUcLftTky948ERqBorY9jLIK4w/exec";


  const input = document.createElement("input");

  input.type = "hidden";

  input.name = "data";

  input.value = dataString;


  form.appendChild(input);

  document.body.appendChild(form);


  form.submit();

}
